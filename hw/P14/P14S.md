    \frac{p_{00} X_i +  p_{01}Y_i + p_{02}Z_i + p_{03}W_i}{p_{20}X_i + p_{21}Y_i + p_{22}Z_i + p_{23}W_i } - x_i

    $$
    \frac{p_{10} X_i +  p_{11}Y_i + p_{12}Z_i + p_{13}W_i}{p_{20}X_i + p_{21}Y_i + p_{22}Z_i + p_{23}W_i } - y_i

    $$
    \text{Adjusted linear residuals:}
    $$
    $$
    p_{00} X_i +  p_{01}Y_i + p_{02}Z_i + p_{03}W_i - x_i (p_{20}X_i + p_{21}Y_i + p_{22}Z_i + p_{23}W_i)

    $$

    $$
